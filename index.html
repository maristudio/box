import { NextResponse } from 'next/server'; // ========================================== // 1. [한-중 번역 브릿지] 헬퍼 함수 // ========================================== async function translateKoToCn(koreanText) { const apiKey = process.env.GEMINI_API_KEY; if (!apiKey) { console.warn("⚠️ [Bridge] GEMINI_API_KEY가 없어 원본 키워드로 진행합니다."); return koreanText; } try { const response = await fetch( `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=${apiKey}`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: `Translate the following shopping/lifestyle keyword to simplified Chinese. Only output the translated word, no other text or explanation: "${koreanText}"` }] }] }) } ); const data = await response.json(); const translated = data.candidates?.[0]?.content?.parts?.[0]?.text?.trim(); return translated || koreanText; } catch (e) { console.error("❌ [Bridge] 번역 실패, 원본 검색어로 우회합니다:", e); return koreanText; } } // ========================================== // 2. 메인 GET 요청 핸들러 // ========================================== export async function GET(request) { try { const { searchParams } = new URL(request.url); const rawKeyword = searchParams.get('keyword') || ''; const platform = searchParams.get('platform') || 'xiaohongshu'; // xiaohongshu 또는 tiktok const sortType = searchParams.get('sort') || 'likes'; // likes, comments, shares const period = searchParams.get('period') || 'all'; // all, weekly, monthly if (!rawKeyword) { return NextResponse.json({ success: false, error: '검색어를 입력해 주세요.' }, { status: 400 }); } // ------------------------------------------ // [번역 브릿지 적용] 한국어 ➔ 중국어 자동 변환 // ------------------------------------------ let searchKeyword = rawKeyword; if (platform === 'xiaohongshu') { searchKeyword = await translateKoToCn(rawKeyword); console.log(`🌐 [Bridge] 번역 완료: "${rawKeyword}" ➔ "${searchKeyword}"`); } // RapidAPI 키 검증 const rapidApiKey = process.env.RAPIDAPI_KEY; if (!rapidApiKey) { throw new Error('RAPIDAPI_KEY가 Vercel 환경 변수에 등록되지 않았습니다.'); } let parsedResults = []; // ========================================== // 3. 샤오홍슈 데이터 수집 및 정밀 파싱 (Parser) // ========================================== if (platform === 'xiaohongshu') { // 본인이 사용하시는 RapidAPI 샤오홍슈 서비스의 정확한 Host와 URL을 적용해 주세요. // (아래는 범용적인 샤오홍슈 search v2 스펙 기준입니다.) const apiHost = 'xiaohongshu-api-v2.p.rapidapi.com'; const url = `https://${apiHost}/api/v2/search?keyword=${encodeURIComponent(searchKeyword)}`; const apiResponse = await fetch(url, { method: 'GET', headers: { 'x-rapidapi-key': rapidApiKey, 'x-rapidapi-host': apiHost } }); if (!apiResponse.ok) { throw new Error(`RapidAPI 통신 실패 (Status: ${apiResponse.status})`); } const resData = await apiResponse.json(); const rawItems = resData.data?.items || []; // 🌟 [핵심] 이중으로 잠긴 'note' 껍데기를 안전하게 분해하는 파싱 로직 parsedResults = rawItems.map(item => { const note = item.note; if (!note) return null; // 빈 껍데기는 버림 const isVideo = note.type === 'video'; // 썸네일 고화질 주소 추출 (여러 백업 필드 확보) const coverImage = note.images_list?.[0]?.url_size_large || note.images_list?.[0]?.url || note.video_info_v2?.image?.thumbnail || ''; // 다운로드용 원본 mp4 동영상 주소 추출 (h264 우선 순위) const playUrl = isVideo ? (note.video_info_v2?.media?.stream?.h264?.[0]?.master_url || note.video_info_v2?.media?.stream?.h265?.[0]?.master_url || '') : ''; return { id: note.id, title: note.title || note.desc?.slice(0, 30) || '제목 없음', desc: note.desc || '', author: note.user?.nickname || 'Unknown', avatar: note.user?.images || '', cover: coverImage, videoUrl: playUrl, likes: note.liked_count || 0, comments: note.comments_count || 0, shares: note.shared_count || 0, timestamp: note.timestamp || 0, // 필터용 타임스탬프 추가 isVideo: isVideo, platform: 'xiaohongshu' }; }).filter(Boolean); // null 데이터 제외 정제 } // ========================================== // 4. 틱톡(TikTok) 수집 라인 분기 (프로젝트 2.0 스펙용) // ========================================== else if (platform === 'tiktok') { // 틱톡의 경우 사용 중인 RapidAPI 스펙에 맞추어 이곳에 파서(Parser)를 빌드인합니다. // (현재는 빈 배열로 반환하되 필요 시 샤오홍슈처럼 파서 확장 가능) parsedResults = []; } // ========================================== // 5. [실시간 기간 필터] 주간 / 월간 처리 // ========================================== if (period !== 'all') { const nowInSeconds = Math.floor(Date.now() / 1000); let limitDuration = 0; if (period === 'weekly') { limitDuration = 7 * 24 * 60 * 60; // 일주일 (초 단위) } else if (period === 'monthly') { limitDuration = 30 * 24 * 60 * 60; // 한 달 (초 단위) } parsedResults = parsedResults.filter(item => { // 데이터에 타임스탬프가 없는 안전장치 적용 if (!item.timestamp) return true; return (nowInSeconds - item.timestamp) <= limitDuration; }); } // ========================================== // 6. [실시간 정렬 필터] 좋아요 / 댓글순 정렬 // ========================================== if (sortType === 'likes') { parsedResults.sort((a, b) => b.likes - a.likes); } else if (sortType === 'comments') { parsedResults.sort((a, b) => b.comments - a.comments); } else if (sortType === 'shares') { parsedResults.sort((a, b) => b.shares - a.shares); } // 🏆 [실시간 수집 한도] 최대 대박 영상 30개 수집 컷 제한 적용 const finalResults = parsedResults.slice(0, 30); // 가뿐히 성공 응답 반환 return NextResponse.json({ success: true, data: finalResults }); } catch (error) { console.error('❌ [API Error] 발굴 실패:', error); // 치명적인 오류가 나서 화면이 아예 굳어버리지 않게 예외 상황에서도 // Fallback(나이키 등의 테스트용) 안전장치 데이터를 전달합니다. return NextResponse.json({ success: false, error: error.message, data: getFallbackData() }, { status: 500 }); } } // ========================================== // 7. [안전장치] 통신 단절 시 하방 보정용 가짜 데이터 // ========================================== function getFallbackData() { return Array.from({ length: 6 }).map((_, i) => ({ id: `fallback-${i}`, title: `[🚨 점검 필요] RapidAPI Key 또는 Host가 올바르지 않습니다.`, desc: 'Vercel Environment Variables 설정에서 RAPIDAPI_KEY 명칭을 다시 대조해 주세요.', author: 'xhs_creator', avatar: '', cover: 'https://images.unsplash.com/photo-1542291026-7eec264c27ff', // 빨간색 나이키 운동화 videoUrl: '', likes: 0, comments: 0, shares: 0, isVideo: false, platform: 'xiaohongshu' })); }
